255  void set_FENE_param(double &le0, double &le1, double &lmin, double &lmax){
256 
257  double width= Average_node_pair_length; // it defines a minimum limit for weil width.
258  double delta_length=Max_node_pair_length-Min_node_pair_length;
259  double width_scaling =0.2; // this variable adjusts the log_barrier potential width. if the edges have almost the same length, then it shlould be redefined to have an appropriate weil width
260  if ((1+ 2*width_scaling)*delta_length < width ){ //this condition shows the case when the mesh is almost ordered.
261  width_scaling= (width-delta_length)/(2*delta_length); //in this case the width tuned in a way that the weil width becomes exactly 0.66*Average_node_pair_lenght
262  // cout<<"\n\nif == true\n\n";
263  }
264  else{
265  width= (1+ 2*width_scaling)*delta_length; //for disordered meshes it sets the weil witdh by scaling factor 0.2. in this case, the width may become larger than 0.66Average which was the minimum limit.
266  // cout<<"\n\nif == false\n\n";
267  }
268  lmin= Min_node_pair_length - width_scaling*delta_length;
269  lmax= Max_node_pair_length + width_scaling*delta_length;
270 
271  le0 = lmin + 3.0*(lmax-lmin)/4.0;
272  le1 = lmin + (lmax-lmin)/4.0;
